        // Dynamic automated price updating

        function recalculateQueuePrices() {

            const selectedPkg = state.packages.find(p => p.id === state.queueForm.packageId);

            if (selectedPkg) {

                const qty = state.queueForm.questionCount || 1;

                state.queueForm.totalPrice = selectedPkg.price * qty;

                document.getElementById('queue-input-total-price').value = state.queueForm.totalPrice;


                if (state.queueForm.paymentStatus === 'ชำระครบแล้ว') {

                    state.queueForm.paidAmount = state.queueForm.totalPrice;

                } else if (state.queueForm.paymentStatus === 'ยังไม่ชำระ') {

                    state.queueForm.paidAmount = 0;

                }

                document.getElementById('queue-input-paid-amount').value = state.queueForm.paidAmount;

            }

            updateQueueBalanceLabel();

            autoSaveFormDraft('queue');

        }

        async function handleTellerSubmit(e) {

                telErr.innerText = 'กรุณากรอกเบอร์โทรศัพท์สำหรับแอดมิน';

                telErr.classList.remove('hidden');

                hasErr = true;

            }

                // Prepare data payload

                const dataPayload = {

                    name: state.tellerForm.name.trim(),

                    pageName: state.tellerForm.pageName.trim(),

                    }

                    if (state.editingTellerId) {

                        state.tellers = state.tellers.map(t => t.id === state.editingTellerId ? { ...dataPayload, id: state.editingTellerId } : t);

                    } else {

                        const docId = 'teller_' + Date.now();

                        state.tellers.push({ ...dataPayload, id: docId });

                    }

            }

        // --- COMPUTED MATH HELPER STATS ENGINE (IN-MEMORY COMPUTATION - RULE 2) ---

        function getComputedMetrics() {

            const filtered = state.queues.filter(q => {

                if (state.currentRole === 'FortuneTeller' && state.currentTellerUser) {

                    return q.tellerId === state.currentTellerUser.id;

                }

                return true;

            });


            const todayStr = getThailandLocalDateOnly(); // GMT+7 Date String

            const thisMonthStr = todayStr.substring(0, 7);


            let pendingPayment = 0, pendingPrediction = 0, predicting = 0, delivered = 0;

            let revToday = 0, revMonth = 0;


            const pkgMap = {};

            const customerMap = {};


            filtered.forEach(q => {

                if (q.jobStatus === 'รอชำระเงิน') pendingPayment++;

                if (q.jobStatus === 'รอคำทำนาย') pendingPrediction++;

                if (q.jobStatus === 'กำลังทำนาย') predicting++;

                if (q.jobStatus === 'ส่งคำทำนายแล้ว') delivered++;


                const paid = parseFloat(q.paidAmount) || 0;

                

                // convert UTC database createdAt back to Thailand timezone to compare safely

                if (q.createdAt) {

                    const localCreatedAt = getThailandLocalDateOnly(new Date(q.createdAt));

                    if (localCreatedAt === todayStr) revToday += paid;

                    if (localCreatedAt.startsWith(thisMonthStr)) revMonth += paid;

                }


                if (q.packageId) pkgMap[q.packageId] = (pkgMap[q.packageId] || 0) + 1;

                if (q.fullname || q.nickname) {

                    const key = q.fullname || q.nickname;

                    customerMap[key] = (customerMap[key] || 0) + 1;

                }

            });


            const popularPkgs = Object.keys(pkgMap).map(id => {

                const pkg = state.packages.find(p => p.id === id);

                return {

                    name: pkg ? pkg.name : 'กำหนดเอง',

                    count: pkgMap[id],

                    revenue: pkgMap[id] * (pkg ? pkg.price : 0)

                };

            }).sort((a, b) => b.count - a.count).slice(0, 3);


            const regulars = Object.keys(customerMap).map(name => {

                const custQueues = filtered.filter(q => (q.fullname || q.nickname) === name);

                const spent = custQueues.reduce((sum, q) => sum + (parseFloat(q.paidAmount) || 0), 0);

                return { name, count: customerMap[name], spent };

            }).sort((a, b) => b.count - a.count).slice(0, 5);


            const newCusts = Object.values(customerMap).filter(v => v === 1).length;

            const returnCusts = Object.values(customerMap).filter(v => v > 1).length;


            const last5Days = Array.from({ length: 5 }, (_, i) => {

                const d = new Date();

                d.setDate(d.getDate() - (4 - i));

                return getThailandLocalDateOnly(d);

            });


            const thaiMonths = ["ม.ค.", "ก.พ.", "มี.ค.", "เม.ย.", "พ.ค.", "มิ.ย.", "ก.ค.", "ส.ค.", "ก.ย.", "ต.ค.", "พ.ย.", "ธ.ค."];

            const chartData = last5Days.map(dateStr => {

                const dayTotal = filtered

                    .filter(q => q.createdAt && getThailandLocalDateOnly(new Date(q.createdAt)) === dateStr)

                    .reduce((sum, q) => sum + (parseFloat(q.paidAmount) || 0), 0);

                const parts = dateStr.split('-');

                const dayLabel = `${parseInt(parts[2])} ${thaiMonths[parseInt(parts[1]) - 1]}`;

                return { label: dayLabel, amount: dayTotal };

            });


            return {

                pendingPayment, pendingPrediction, predicting, delivered,

                revToday, revMonth, popularPkgs, regulars, newCusts, returnCusts, chartData

            };

        }

                                queueId: q.id

                            });

                        } else if (diffHrs <= 0) {

                            alerts.push({

                                type: 'danger',

                                title: 'เลยกำหนดส่ง ⚠️',

                        }

                    } catch (e) {}

                }

            });


            return alerts;

        }

        function openConfirmDialog(title, message, onOkCallback) {

            document.getElementById('confirm-title').innerText = title;

            document.getElementById('confirm-message').innerText = message;

            

            const okBtn = document.getElementById('confirm-ok');

            // Clone button to clear previous click event listeners cleanly

            const newOkBtn = okBtn.cloneNode(true);

            okBtn.parentNode.replaceChild(newOkBtn, okBtn);


            newOkBtn.addEventListener('click', () => {

                onOkCallback();

                document.getElementById('confirm-modal').classList.add('hidden');

            });


            document.getElementById('confirm-modal').classList.remove('hidden');

        }


        function showToast(message, type = 'success') {

            const id = Date.now();

            const container = document.getElementById('toasts-container');

            

            const toastDiv = document.createElement('div');

            toastDiv.id = `toast-${id}`;

            toastDiv.className = `flex items-center justify-between p-4 rounded-xl shadow-lg border animate-slide-in text-sm font-medium transition-all pointer-events-auto ${

                type === 'error' 

                    ? 'bg-rose-50 border-rose-100 text-rose-800 dark:bg-rose-950/90 dark:border-rose-900/40 dark:text-rose-200' 

                    : 'bg-emerald-50 border-emerald-100 text-emerald-800 dark:bg-emerald-950/90 dark:border-emerald-900/40 dark:text-emerald-200'

            }`;

            

            toastDiv.innerHTML = `

                <span>${message}</span>

                <button onclick="dismissToast(${id})" class="ml-2 text-zinc-400 hover:text-zinc-600 dark:hover:text-zinc-200">

                    <i class="fa-solid fa-xmark"></i>

                </button>

            `;


            container.appendChild(toastDiv);


            setTimeout(() => {

                dismissToast(id);

            }, 4000);

        }
